# syntax:
#   filename-match pattern : command
# Use '%' to insert filename
*.nix: alejandra %

# Patterns separated with ; will trigger if any are matched
*.sh;*.bash: shfmt -i 4 -ci -s -bn -w %

# Newlines are also valid pattern separators
*.c
*.h
*.cpp
*.hpp
    : clang-format -i %
